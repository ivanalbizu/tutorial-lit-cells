dev — Vite dev server con HMR
build — TypeScript + Rollup para producción
preview — previsualizar el build

package.json

Añadimos dependencias

tsconfig.json

Las dos opciones críticas para Lit


{
  "compilerOptions": {
    "target": "ES2021",
    "module": "ESNext",
    "moduleResolution": "bundler",
    "lib": ["ES2021", "DOM", "DOM.Iterable"],
    "strict": true,
    "esModuleInterop": true,
    "skipLibCheck": true,
    "forceConsistentCasingInFileNames": true,
    "useDefineForClassFields": false,
    "experimentalDecorators": true,
    "declaration": true,
    "sourceMap": true,
    "outDir": "dist",
    "rootDir": "src"
  },
  "include": ["src"]
}

experimentalDecorators: true
Lit usa @customElement, @property...

useDefineForClassFields: false
Sin esto, las propiedades reactivas NO funcionan

vite.config.ts

Configuración mínima de Vite


import { defineConfig } from 'vite';

export default defineConfig({
  build: {
    lib: {
      entry: 'src/index.ts',
      formats: ['es'],
    },
    rollupOptions: {
      external: /^lit/,
    },
  },
});

external: /^lit/ — no incluir Lit en el bundle de producción
